27        id: "Switch1".to_string(),
28    }
29    .into();

32
33    let cmd = TurnLightSwitchOff {
34        id: "Switch1".to_string(),
35    }
36    .into();
37    command::Handler::handle(&svc, cmd).await?;
38    println!("Turned Switch1 Off");
39
40    let query = GetSwitchState {
41        id: "Switch1".to_string(),
42    }
43    .into();
44    let state = query::Handler::handle(&svc, query).await?;

47}
